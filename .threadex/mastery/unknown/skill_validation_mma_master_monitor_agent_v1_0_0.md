    """
    Calculate weighted MMA score from 7D scores.
    
    Args:
        scores: Dict with keys D1-D7, values 1-10
    
    Returns:
        Weighted average (0-10)
    """
    weights = {
        "D1": 0.15,  # Strategy Alignment
        "D2": 0.20,  # Proof Discipline (highest)
        "D3": 0.15,  # CTA Integrity
        "D4": 0.15,  # Voice Consistency
        "D5": 0.10,  # Clarity + Structure
        "D6": 0.15,  # Resonance
        "D7": 0.10,  # Ethical Guardrails
    }

    """
    Score how well content covers all 7S dimensions.
    """
    seven_s_markers = {
        "SAFE": ["guarantee", "refund", "risk-free", "proven",

        "SPECIAL": ["exclusive", "unique", "only", "special",

        "SUPERIOR": ["welcome", "belong", "family", "together",

        "STEAL": ["value", "worth", "bonus", "free", "included",
                  "savings", "deal"]
    }
    
    text_lower = text.lower()
    scores = {}

        hits = sum(1 for m in markers if m in text_lower)

    return {
        "scores": scores,
