### 5.2 Memory Architecture (Four Types)

**1. EPISODIC MEMORY** (Specific Interactions)
```sql
-- Table: episodic_memory (Convex/Firebase)
CREATE TABLE episodic_memory (
  id UUID PRIMARY KEY,
  conversation_id UUID,
  user_id UUID,
  
  -- Interaction
  user_message TEXT,
  assistant_response TEXT,
  neurobox_activation JSONB, -- Which dimensions activated
  
  -- Context
  user_sentiment TEXT,
  urgency_level INTEGER,
  topic_classification TEXT[],
  
  -- Outcome
  user_satisfaction INTEGER,
  follow_up_question BOOLEAN, -- Sign of incomplete response
  
  created_at TIMESTAMPTZ,
  embedding VECTOR(1536) -- Semantic search
);
```

**2. SEMANTIC MEMORY** (Facts & Knowledge)
```sql
-- Table: semantic_memory
CREATE TABLE semantic_memory (
  id UUID PRIMARY KEY,
  
  -- Knowledge
  concept TEXT,
  definition TEXT,
  source TEXT,
  confidence_score NUMERIC(3,2),
  
  -- Relationships
  related_concepts TEXT[],
  contradicts TEXT[],
  
  -- Evolution
  version INTEGER,
  supersedes UUID REFERENCES semantic_memory(id),
  
  created_at TIMESTAMPTZ,
  embedding VECTOR(1536)
);
```

**3. PROCEDURAL MEMORY** (How-To Skills)

### 5.3 Self-Healing Edge Function

```typescript
// Convex edge function: selfHealingMonitor.ts
import { mutation, query } from './_generated/server';

export const detectAndHeal = mutation(async ({ db }) => {
  // 1. Analyze recent reflections (last 20 interactions)
  const recentReflections = await db
    .query('reflections')
    .order('desc')
    .take(20);
  
  // 2. Check against known pattern library
  const activePatterns = await db
    .query('pattern_memory')
    .filter(q => q.eq('is_active', true))
    .collect();
  
  for (const pattern of activePatterns) {
    const occurrences = countPatternOccurrences(
      recentReflections,
      pattern.detection_signature
    );
    
    // 3. Check if threshold reached
    if (occurrences >= pattern.detection_threshold) {
      console.log(`🔧 HEALING: Pattern ${pattern.pattern_type} detected`);
      
      // 4. Apply healing action
      switch (pattern.healing_action_type) {
        case 'skill_patch':
          await applySkillPatch(pattern.skill_patch_id);
          break;
        case 'constitutional_patch':
          await applyConstitutionalPatch(pattern.constitutional_patch_id);
          break;
        case 'emergency_rollback':
          await emergencyRollback(pattern.rollback_to_version);
          break;
      }
      
      // 5. Log healing event
      await db.insert('healing_events', {
        pattern_id: pattern._id,
        action_taken: pattern.healing_action,
        triggered_at: Date.now(),
        triggered_by_reflections: recentReflections.slice(0, 3).map(r => r._id)
      });
      
      // 6. Update pattern stats
      await db.patch(pattern._id, {
        times_healed: pattern.times_healed + 1
      });
    }
  }
});

// Helper: Count pattern occurrences in recent reflections
function countPatternOccurrences(reflections: any[], signature: any): number {
  let count = 0;
  
  for (const reflection of reflections) {
    const matches = checkSignatureMatch(reflection, signature);
    if (matches) count++;
  }
  
  return count;
}

// Helper: Check if reflection matches pattern signature
function checkSignatureMatch(reflection: any, signature: any): boolean {
  // Example signature: { dimension: "SMART", score_range: [1, 2], keywords: [...] }
  
  if (signature.dimension && reflection.neurobox_scores) {
    const score = reflection.neurobox_scores[signature.dimension];
    if (score < signature.score_range[0] || score > signature.score_range[1]) {
      return false;
    }
  }
  
  if (signature.keywords && reflection.content) {
    const hasKeyword = signature.keywords.some((kw: string) => 
      reflection.content.toLowerCase().includes(kw.toLowerCase())
    );
    if (!hasKeyword) return false;
  }
  
  return true;
}
```
